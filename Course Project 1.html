activity2$date <- as.Date(as.character(activity2$date), "%Y-%m-%d")
activity2$day <- weekdays(activity2$date)
activity2$daytype <- ifelse(activity2$day == "Saturday"| activity2$day == "Sunday", "weekend", "weekday")
